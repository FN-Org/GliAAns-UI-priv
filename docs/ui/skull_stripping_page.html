295    def has_existing_skull_strip(self, nifti_file_path, workspace_path):
296        """
297        Check if a skull-stripped file already exists for the selected subject.
298
299        Args:
300            nifti_file_path (str): Full path to input NIfTI file.
301            workspace_path (str): Base workspace directory path.
302
303        Returns:
304            bool: True if an existing skull strip is found, False otherwise.
305        """
306        path_parts = nifti_file_path.replace(workspace_path, '').strip(os.sep).split(os.sep)
307        subject_id = next((p for p in path_parts if p.startswith('sub-')), None)
308        if not subject_id:
309            return False
310
311        skull_strip_dir = os.path.join(workspace_path, 'derivatives', 'skullstrips', subject_id, 'anat')
312        if not os.path.exists(skull_strip_dir):
313            return False
314
315        return any(f.endswith('.nii.gz') for f in os.listdir(skull_strip_dir))

323    def run_bet(self):
324        """Start the skull stripping process in a background thread."""
325        selected_files = self.file_selector_widget.get_selected_files()
326        if not selected_files:
327            QMessageBox.warning(self, "No files", "Please select at least one NIfTI file first.")
328            return
329
330        parameters = None
331        if self.has_bet:
332            parameters = {
333                'f_val': self.f_spinbox.value(),
334                'opt_brain_extracted': self.opt_brain_extracted.isChecked(),
335                'opt_m': self.opt_m.isChecked(),
336                'opt_t': self.opt_t.isChecked(),
337                'opt_s': self.opt_s.isChecked(),
338                'opt_o': self.opt_o.isChecked(),
339                'g_val': self.g_spinbox.value(),
340                'c_x': self.c_x_spinbox.value(),
341                'c_y': self.c_y_spinbox.value(),
342                'c_z': self.c_z_spinbox.value(),
343            }
344
345        if self.has_bet:
346            self.bet_tool = "fsl-bet"
347        elif self.has_synthstrip:
348            self.bet_tool = "synthstrip"
349        else:
350            self.bet_tool = "hd-bet"
351
352        # Create worker thread
353        self.worker = SkullStripThread(selected_files, self.context["workspace_path"], parameters, self.has_cuda, self.bet_tool)
354
355        # Connect worker signals
356        self.worker.progress_updated.connect(self.on_progress_updated)
357        self.worker.progress_value_updated.connect(self.on_progress_value_updated)
358        self.worker.file_started.connect(self.on_file_started)
359        self.worker.file_completed.connect(self.on_file_completed)
360        self.worker.all_completed.connect(self.on_all_completed)
361        self.worker.finished.connect(self.on_worker_finished)
362
363        # Update UI
364        self.processing.emit(True)
365        self.progress_bar.setRange(0, 100)
366        self.progress_bar.setValue(0)
367        self.progress_bar.setVisible(True)
368
369        self.worker.start()

296    def has_existing_skull_strip(self, nifti_file_path, workspace_path):
297        """
298        Check if a skull-stripped file already exists for the selected subject.
299
300        Args:

324    def run_bet(self):
325        """Start the skull stripping process in a background thread."""
326        selected_files = self.file_selector_widget.get_selected_files()
327        if not selected_files:
328            QMessageBox.warning(self, "No files", "Please select at least one NIfTI file first.")

348        elif self.has_synthstrip:

350        else:
351            self.bet_tool = "hd-bet"
