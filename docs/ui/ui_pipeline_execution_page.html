 90    def _find_latest_config(self):
 91        """Find the config file with the highest numeric ID in the pipeline directory."""
 92        import glob
 93        pipeline_dir = os.path.join(self.workspace_path, "pipeline")
 94
 95        if not os.path.exists(pipeline_dir):
 96            return os.path.join(pipeline_dir, "pipeline_config.json")
 97
 98        config_pattern = os.path.join(pipeline_dir, "*_config.json")
 99        config_files = glob.glob(config_pattern)
100
101        if not config_files:
102            return os.path.join(pipeline_dir, "pipeline_config.json")
103
104        max_id = 0
105        latest_config = None
106
107        # Parse filenames to detect the latest config ID
108        for config_file in config_files:
109            filename = os.path.basename(config_file)
110            try:
111                config_id = int(filename.split('_')[0])
112                if config_id > max_id:
113                    max_id = config_id
114                    latest_config = config_file
115            except (ValueError, IndexError):
116                continue
117
118        return latest_config or os.path.join(pipeline_dir, "pipeline_config.json")

243    def _start_pipeline(self):
244        """Start the external pipeline process and monitor its output."""
245        if self.pipeline_process:
246            return  # Already running
247
248        self.pipeline_process = QProcess()
249        self.pipeline_process.finished.connect(self._on_process_finished)
250        self.pipeline_process.errorOccurred.connect(self._on_process_error)
251        self.pipeline_process.readyReadStandardOutput.connect(self._on_stdout_ready)
252        self.pipeline_process.readyReadStandardError.connect(self._on_stderr_ready)
253
254        self.progress_bar.setValue(0)
255        self.stop_button.setEnabled(True)
256        self._log_message("Starting pipeline execution...")
257
258        cmd = [
259            self.pipeline_bin_path,
260            "--config", self.config_path,
261            "--work-dir", self.workspace_path,
262            "--out-dir", self.pipeline_output_dir
263        ]
264        log.debug("Executing: " + " ".join(cmd))
265        self.pipeline_process.start(cmd[0], cmd[1:])
266
267        if not self.pipeline_process.waitForStarted(3000):
268            self._on_pipeline_error("Failed to start pipeline process")

273    def _on_process_finished(self, exit_code, exit_status):
274        """Called when the process terminates."""
275        if exit_code == 0 and exit_status == QProcess.ExitStatus.NormalExit:

280    def _on_process_error(self, error):
281        """Handle QProcess errors."""
282        error_messages = {
283            QProcess.ProcessError.FailedToStart: "Failed to start process",
284            QProcess.ProcessError.Crashed: "Process crashed",
285            QProcess.ProcessError.Timedout: "Process timed out",
286            QProcess.ProcessError.WriteError: "Write error",
287            QProcess.ProcessError.ReadError: "Read error",
288            QProcess.ProcessError.UnknownError: "Unknown error"
289        }
290        msg = error_messages.get(error, f"Process error: {error}")
291        self._on_pipeline_error(msg)

296    def _on_stdout_ready(self):
297        """Triggered when stdout has new data."""
298        if not self.pipeline_process:
299            return
300        output = bytes(self.pipeline_process.readAllStandardOutput()).decode().strip()
301        for line in output.split('\n'):
302            if line.strip():
303                self._process_pipeline_output(line.strip())

305    def _on_stderr_ready(self):
306        """Triggered when stderr has new data."""
307        if not self.pipeline_process:
308            return
309        output = bytes(self.pipeline_process.readAllStandardError()).decode().strip()
310        for line in output.split('\n'):
311            if line.strip():
312                self._log_message(f"STDERR: {line.strip()}")

317    def _process_pipeline_output(self, line):
318        """Interpret and act upon a single line of pipeline output."""
319        if line.startswith("LOG: "):
320            self._log_message(line[5:])
321            self._update_current_operation(line[5:])
322        elif line.startswith("ERROR: "):
323            self._log_message("ERROR: " + line[7:])
324        elif line.startswith("PROGRESS: "):
325            self._update_progress(line[10:])
326        elif line.startswith("FINISHED: "):
327            self._log_message("Pipeline finished.")
328        else:
329            self._log_message(line)

360        """

363        Expected format from pipeline: `"PROGRESS: X/Y"`

365        Args:
366            progress_info (str): Progress data string (e.g. `"3/10"`).

369            if '/' in progress_info:
370                current, total = map(int, progress_info.split('/'))
371                percentage = int((current / total) * 100)
372                self.progress_bar.setValue(percentage)

402            self.current_operation.setText(

411        """
412        Extract a patient ID (e.g. 'sub-001') from a log message.

414        Args:
415            message (str): Log message.

451    def _on_stop_clicked(self):
452        """
453        Handle user click on the "Stop Pipeline" button.
454
455        Attempts a graceful termination, then a forced kill if needed.
456        """
457        if self.pipeline_process and self.pipeline_process.state() == QProcess.ProcessState.Running:
458            self._log_message("Stopping pipeline...")
459
460            # Try graceful termination first
461            self.pipeline_process.terminate()
462
463            # Force-kill if not stopped within 5 seconds
464            if not self.pipeline_process.waitForFinished(5000):
465                self.pipeline_process.kill()
466                self.pipeline_process.waitForFinished(3000)
467
468            self._log_message("Pipeline stopped by user.")
469            self.current_operation.setText("Execution was interrupted by user.")
470            self.progress_bar.setValue(0)
471            self.stop_button.setEnabled(False)
472
473            self.pipeline_process = None
474            self.context["update_main_buttons"]()

 91    def _find_latest_config(self):
 92        """Find the config file with the highest numeric ID in the pipeline directory."""
 93        import glob
 94        pipeline_dir = os.path.join(self.workspace_path, "pipeline")
 95
 96        if not os.path.exists(pipeline_dir):
 97            return os.path.join(pipeline_dir, "pipeline_config.json")
 98
 99        config_pattern = os.path.join(pipeline_dir, "*_config.json")
100        config_files = glob.glob(config_pattern)
101
102        if not config_files:
103            return os.path.join(pipeline_dir, "pipeline_config.json")
104
105        max_id = 0
106        latest_config = None
107
108        # Parse filenames to detect the latest config ID
109        for config_file in config_files:
110            filename = os.path.basename(config_file)
111            try:
112                config_id = int(filename.split('_')[0])
113                if config_id > max_id:
114                    max_id = config_id
115                    latest_config = config_file
116            except (ValueError, IndexError):
117                continue
118
119        return latest_config or os.path.join(pipeline_dir, "pipeline_config.json")
