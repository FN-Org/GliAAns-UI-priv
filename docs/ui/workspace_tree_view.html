 67    def handle_workspace_click(self):
 68        """Handle single-click event: update selected files and emit them via context signal."""
 69        selected_indexes = self.selectionModel().selectedRows()
 70        selected_files = [self.tree_model.filePath(idx) for idx in selected_indexes]

 75    def handle_double_click(self, index):
 76        """
 77        Handle double-click event:
 78        - If the selected item is a NIfTI file (.nii / .nii.gz), open it in the viewer.
 79        - Otherwise, open it with the system's default application.
 80        """
 81        file_path = self.tree_model.filePath(index)
 82        if not os.path.isfile(file_path):
 83            return

 86            try:

 88            except Exception as e:
 89                QMessageBox.critical(
 90                    self,
 91                    QtCore.QCoreApplication.translate("TreeView", "Error"),
 92                    QtCore.QCoreApplication.translate("TreeView", "Error when opening NIfTI file:\n{0}").format(str(e))
 93                )
 94                log.error(f"Error opening NIfTI file:\n{str(e)}")
 95        else:
 96            self._open_in_explorer(file_path)

112    def open_tree_context_menu(self, position):
113        """
114        Open the right-click context menu dynamically, depending on the selected item:
115        - Workspace root
116        - Folder
117        - File
118        - Multiple files
119        """
120        index = self.indexAt(position)
121        file_path = None
122        is_dir = False
123        is_nifty = False
124
125        # Determine selected file/folder
126        if self.selected_files and len(self.selected_files) == 1 and not index.isValid():
127            file_path = self.selected_files[0]
128            is_dir = self.tree_model.isDir(index) if index.isValid() else False
129            is_nifty = file_path.endswith((".nii", ".nii.gz"))
130        elif index.isValid():
131            file_path = self.tree_model.filePath(index)
132            is_dir = self.tree_model.isDir(index)
133
134        menu = QMenu(self)
135        actions = {}
136
137        # Choose menu layout based on selection
138        if not index.isValid() and not self.selected_files:
139            actions.update(self._add_workspace_actions(menu))
140        elif index.isValid() and file_path and len(self.selected_files) <= 1:
141            if is_dir:
142                actions.update(self._add_folder_actions(menu, file_path))
143            else:
144                actions.update(self._add_file_actions(menu, file_path, is_nifty))
145        elif index.isValid() and len(self.selected_files) > 1:
146            actions.update(self._add_multi_file_actions(menu))
147
148        # Execute chosen action
149        chosen_action = menu.exec(self.viewport().mapToGlobal(position))
150        if chosen_action and chosen_action in actions:
151            actions[chosen_action](file_path, is_dir, is_nifty)

222    def add_file_to_workspace(self, folder_path, is_dir):
223        """
224        Add a single file into the workspace.
225        Opens a file dialog, optionally adds related JSON, and triggers background copy thread.
226        """
227        dialog = QFileDialog(self.context['main_window'], "Select File")
228        dialog.setFileMode(QFileDialog.FileMode.ExistingFile)
229        dialog.setOption(QFileDialog.Option.DontUseNativeDialog, True)
230        dialog.setOption(QFileDialog.Option.ReadOnly, True)
231        dialog.setDirectory(os.path.expanduser("~"))
232
233        if not dialog.exec():
234            return
235
236        file = dialog.selectedFiles()[0]
237        file_name = os.path.basename(file)
238        dir_name = os.path.dirname(file)
239        name, ext = os.path.splitext(file_name)
240        json_file = None
241
242        # Handle potential .nii.gz format
243        if ext == ".gz":
244            name2, ext2 = os.path.splitext(name)
245            if ext2 == ".nii":
246                json_file = self._including_json(name2, dir_name)
247        elif ext == ".nii":
248            json_file = self._including_json(name, dir_name)
249
250        # Determine destination folder
251        if folder_path:
252            if is_dir:
253                folder = os.path.basename(folder_path)
254            else:
255                folder = os.path.dirname(folder_path)
256        else:
257            folder = "workspace"
258
259        # Specialized behavior based on folder type
260        if folder in ("anat", "pet"):
261            self.new_thread.emit(CopyDeleteThread(src=file, dst=folder_path, is_folder=is_dir, copy=True))
262        elif re.match(r"^ses-\d+$", folder):
263            self.new_thread.emit(CopyDeleteThread(src=file, dst=os.path.join(folder_path, "pet"), is_folder=is_dir, copy=True))
264        elif re.match(r"^sub-\d+$", folder):
265            return self.open_role_dialog(files=[file, json_file], folder_path=folder_path, subj=folder)
266        elif folder == "derivatives":
267            return self.open_role_dialog(files=[file, json_file], folder_path=folder_path, main=folder)
268        else:
269            return self.open_role_dialog(files=[file, json_file], folder_path=self.workspace_path)

271    def open_role_dialog(self, files, folder_path=None, subj=None, role=None, main=None):
272        """Open a FileRoleDialog to determine where to place files in the workspace hierarchy."""
273        dialog = FileRoleDialog(workspace_path=self.workspace_path, subj=subj, role=role, main=main, parent=self)
274        if dialog.exec():
275            relative_path = dialog.get_relative_path()
276            path = os.path.join(folder_path, relative_path)
277            os.makedirs(path, exist_ok=True)
278            for file in files:
279                if file:
280                    self.new_thread.emit(CopyDeleteThread(src=file, dst=path, is_folder=False, copy=True))

304    def export_files(self, paths, is_dir):
305        """Export selected files or folders to a chosen destination."""
306        if not paths:
307            return
308
309        def _get_file_filter(ext):
310            """Return the QFileDialog filter based on file extension."""
311            if ext == ".nii.gz":
312                return "NIfTI compressed (*.nii.gz);;NIfTI (*.nii);;All files (*.*)"
313            elif ext == ".nii":
314                return "NIfTI (*.nii);;All files (*.*)"
315            elif ext == ".json":
316                return "JSON (*.json);;All files (*.*)"
317            return "All files (*.*)"
318
319        def _get_json_file(path):
320            """Return associated JSON file if it exists."""
321            base, ext = os.path.splitext(path)
322            if ext == ".gz" and base.endswith(".nii"):
323                base2 = base[:-4]  # remove .nii
324                json_file = os.path.join(os.path.dirname(path), base2 + ".json")
325                return json_file if os.path.exists(json_file) else None
326            elif ext == ".nii":
327                json_file = os.path.join(os.path.dirname(path), base + ".json")
328                return json_file if os.path.exists(json_file) else None
329            return None
330
331        if len(paths) == 1:
332            path = paths[0]
333            if is_dir:
334                # Choose destination folder for a single directory
335                dst_path = QFileDialog.getExistingDirectory(self, "Select destination folder")
336                if dst_path:
337                    dst_path = os.path.join(dst_path, os.path.basename(path))
338                    self.new_thread.emit(CopyDeleteThread(src=path, dst=dst_path, is_folder=True, copy=True))
339            else:
340                # Single file export
341                file_filter = _get_file_filter(os.path.splitext(path)[1])
342                dst_path, _ = QFileDialog.getSaveFileName(self, "Save file as...", os.path.basename(path), file_filter)
343                if not dst_path:
344                    return
345
346                self.new_thread.emit(CopyDeleteThread(src=path, dst=dst_path, is_folder=False, copy=True))
347
348                # Copy associated JSON if exists
349                json_file = _get_json_file(path)
350                if json_file:
351                    dst_json = os.path.splitext(dst_path)[0] + ".json"
352                    self.new_thread.emit(CopyDeleteThread(src=json_file, dst=dst_json, is_folder=False, copy=True))
353        else:
354            # Multiple files/folders
355            dst_dir = QFileDialog.getExistingDirectory(self, "Select destination folder")
356            if not dst_dir:
357                return
358
359            for path in paths:
360                if not os.path.exists(path):
361                    continue
362
363                is_folder = os.path.isdir(path)
364                dst_path = os.path.join(dst_dir, os.path.basename(path)) if is_folder else os.path.join(dst_dir,
365                                                                                                        os.path.basename(
366                                                                                                            path))
367                self.new_thread.emit(CopyDeleteThread(src=path, dst=dst_path, is_folder=is_folder, copy=True))

369    def remove_from_workspace(self, paths):
370        """Ask confirmation before deleting selected files/folders from workspace."""
371        for path in paths:
372            if not os.path.exists(path):
373                return
374        if len(paths)<2:
375            is_dir = os.path.isdir(paths[0])
376            item_type = QtCore.QCoreApplication.translate("TreeView", "this folder: ") if is_dir else QtCore.QCoreApplication.translate("TreeView", "this file:")
377        else: item_type = QtCore.QCoreApplication.translate("TreeView", "these files")
378
379        item_name = os.path.basename(paths[0]) if len(paths) == 1 else ""
380        message = QtCore.QCoreApplication.translate("TreeView", "Are you sure you want to remove {0} {1}?").format(item_type, item_name)
381
382        reply = QMessageBox.question(
383            self,
384            QtCore.QCoreApplication.translate("TreeView", "Confirm"),
385            message,
386            QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No
387        )
388
389        if reply == QMessageBox.StandardButton.Yes:
390            for path in paths:
391                is_dir = os.path.isdir(path)
392                self.new_thread.emit(CopyDeleteThread(src=path, is_folder=is_dir,delete=True))
