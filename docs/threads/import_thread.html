 88        """
 89        Main thread entry point.
 90
 91        This method orchestrates the following high-level behaviors:
 92          - If a single folder is provided:
 93              * Detect BIDS folder and copy directly
 94              * Else detect whether folder contains multiple patient subfolders or
 95                a single subject split across many DICOM series
 96              * Convert found DICOM series to NIfTI using dcm2niix in a temporary
 97                directory, then convert NIfTI + JSON to a BIDS-like structure inside
 98                the workspace.
 99          - If multiple folders are provided:
100              * Process each folder independently with `_handle_import`.
101        Progress is emitted periodically. Exceptions are forwarded via the
102        `error` signal.
103        """
104        try:
105            # initial small progress step
106            self.current_progress = 10
107            self.progress.emit(self.current_progress)
108
109            # Validate input list length and branch accordingly
110            if len(self.folders_path) == 1:
111                # single path handling
112                folder_path = self.folders_path[0]
113
114                # Basic validation
115                if not os.path.isdir(folder_path):
116                    raise Exception(QCoreApplication.translate("Threads", "Invalid folders path"))
117
118                # Case A: already a BIDS subject folder -> copy directly
119                if self._is_bids_folder(folder_path):
120                    log.debug(f"BIDS structure detected in: {folder_path}")
121
122                    if self._is_canceled:
123                        return
124                    self.current_progress = 50
125                    self.progress.emit(self.current_progress)
126
127                    # choose next sub-id (e.g. sub-03) and copy tree
128                    old_sub_id = os.path.basename(os.path.normpath(folder_path))
129                    new_sub_id = self._get_next_sub_id()
130                    dest = os.path.join(self.workspace_path, new_sub_id)
131                    shutil.copytree(folder_path, dest, dirs_exist_ok=True)
132
133                    if not self._is_canceled:
134                        self._rename_bids_files(dest, old_sub_id, new_sub_id)
135
136                    if self._is_canceled:
137                        return
138                    self.current_progress = 100
139                    self.progress.emit(self.current_progress)
140
141                    log.debug(f"BIDS folder copied and renamed to {new_sub_id}.")  # <-- Log aggiornato
142                    self.finished.emit()
143                    return
144
145                # Case B: folder contains only subfolders (possible multi-patient)
146                subfolders = [
147                    os.path.join(folder_path, d)
148                    for d in os.listdir(folder_path)
149                    if os.path.isdir(os.path.join(folder_path, d))
150                ]
151
152                # If directory contains just subfolders and no direct nifti/dicom files, decide:
153                # - If subfolders belong to a single subject (multiple series in separate folders),
154                #   treat whole folder as one subject.
155                if subfolders and not any(
156                        self._is_nifti_file(f) or self._is_dicom_file(os.path.join(folder_path, f))
157                        for f in os.listdir(folder_path)
158                ):

161                    # process entire root as a single patient (e.g., MR + PET series in different
162                    # DICOM folder exports).

164                        log.debug(f"Single subject spread across {len(subfolders)} DICOM folders in: {folder_path}")
165
166                        # Process the root folder as single subject
167                        self._process_single_patient_folder(folder_path)
168                        if self._is_canceled:
169                            return
170                        self.progress.emit(100)
171                        self.finished.emit()
172                        return
173
174                    # Otherwise treat each subfolder as separate patient and import them
175                    log.debug(f"Multiple patient folders detected in: {folder_path}")
176                    folders_num = len(subfolders)
177                    progress_for_folder = 90 / folders_num
178                    self.current_progress = 10
179
180                    for subfolder in subfolders:
181                        if self._is_canceled:
182                            return
183                        self._handle_import(subfolder)
184                        if self._is_canceled:
185                            return
186                        self.current_progress += progress_for_folder
187                        self.progress.emit(self.current_progress)
188
189                    if self._is_canceled:
190                        return
191                    self.progress.emit(100)
192                    self.finished.emit()
193                    return

195                # Case C: folder may contain files (NIfTI, DICOM, misc) -> collect and convert
196                nifti_files = []
197                dicom_files = []
198
199                base_folder_name = os.path.basename(os.path.normpath(folder_path))
200
201                # create temporary directory for conversion and intermediate files
202                temp_dir = tempfile.mkdtemp()
203                temp_base_dir = os.path.join(temp_dir, base_folder_name)
204
205                # Walk the folder, copy non-medical files into the temp tree and list nifti/dicom
206                for root, _, files in os.walk(folder_path):
207                    for file in files:
208                        if self._is_canceled:
209                            # cleanup and exit early when cancelled
210                            shutil.rmtree(temp_dir, ignore_errors=True)
211                            return
212                        file_path = os.path.join(root, file)
213
214                        # Keep folder structure inside temp_base_dir
215                        relative_path = os.path.relpath(root, folder_path)
216                        dest_dir = os.path.join(temp_base_dir, relative_path)
217                        os.makedirs(dest_dir, exist_ok=True)
218
219                        if self._is_nifti_file(file):
220                            # record source and destination pair for later copy
221                            nifti_files.append((file_path, os.path.join(dest_dir, file)))
222                        elif self._is_dicom_file(file_path):
223                            # record dicom file for conversion
224                            dicom_files.append(file_path)
225                        else:
226                            # other files (sidecars, metadata, reports) are copied directly
227                            shutil.copy2(file_path, os.path.join(dest_dir, file))
228                            log.debug(f"Imported other file: {os.path.join(relative_path, file)}")
229
230                # Update progress after scan
231                self.current_progress = 20
232                self.progress.emit(self.current_progress)
233
234                # Copy any found NIfTI files into temp structure and update progress incrementally
235                nifti_num = len(nifti_files)
236                if nifti_num > 0:
237                    progress_per_nifti = int(40 / nifti_num)
238                    for src, dest in nifti_files:
239                        if self._is_canceled:
240                            shutil.rmtree(temp_dir, ignore_errors=True)
241                            return
242                        shutil.copy2(src, dest)
243                        self.current_progress += progress_per_nifti
244                        self.progress.emit(self.current_progress)
245                        log.debug(f"Imported NIfTI file: {os.path.relpath(dest, temp_base_dir)}")
246
247                # Midpoint progress
248                self.current_progress = 60
249                self.progress.emit(self.current_progress)
250
251                if self._is_canceled:
252                    return
253
254                # If DICOM files were found, convert the original folder to NIfTI into the temp tree
255                if dicom_files:
256                    self._convert_dicom_folder_to_nifti(folder_path, temp_base_dir)
257
258                # small progress bump
259                self.current_progress += 10
260                self.progress.emit(self.current_progress)
261
262                # Now transform the temp folder into a BIDS-like structure under workspace
263                self._convert_to_bids_structure(temp_base_dir)
264
265                # Cleanup temporary directory after conversion/copy
266                shutil.rmtree(temp_dir, ignore_errors=True)
267
268            elif len(self.folders_path) == 0:
269                # empty input - emit an error
270                raise Exception(QCoreApplication.translate("Threads", "Invalid folders path"))
271            elif len(self.folders_path) > 1:
272                # multiple root paths - handle each independently
273                progress_for_folder = int(90 / len(self.folders_path))
274                for path in self.folders_path:
275                    if self._is_canceled:
276                        return
277                    self._handle_import(path)
278                    self.current_progress += progress_for_folder
279                    self.progress.emit(self.current_progress)
280            else:
281                # fallback guard - shouldn't be reached
282                raise Exception(QCoreApplication.translate("Threads", "Invalid folders path"))
283
284            # Finalize progress and signal completion
285            self.current_progress = 100
286            self.progress.emit(self.current_progress)
287            self.finished.emit()
288
289        except Exception as e:
290            # If cancellation was requested we avoid emitting an error
291            if self._is_canceled:
292                log.info("Import interrupted by user, no error emitted.")
293                return
294            # Otherwise, emit error and log it
295            self.error.emit(str(e))
296            log.error(f"Error: {str(e)}")

299    def _handle_import(self, folder_path):
300        """
301        Handle the import of a single folder (which may itself contain subfolders).
302
303        This method performs heuristics to determine what kind of folder we are dealing with:
304          1. A BIDS folder â€” copy it directly.
305          2. A folder with direct medical files (NIfTI or DICOM) â€” process as one patient.
306          3. A folder containing subfolders â€” analyze if they belong to one or more patients.
307             - If subfolders are DICOM series of the same patient, process as one.
308             - If subfolders look like different patients (e.g., named 'sub-01', 'patient_02'), process each separately.
309             - Otherwise, default to single-patient mode.
310
311        Args:
312            folder_path (str): Path to the folder to process.
313        """
314        if not os.path.isdir(folder_path):
315            return
316        if self._is_canceled:
317            return
318
319        log.debug(f"Processing folder: {folder_path}")
320
321        # Case 1: The folder is already BIDS-organized -> just copy
322        if self._is_bids_folder(folder_path):
323            log.debug(f"BIDS structure detected in: {folder_path}")
324
325            old_sub_id = os.path.basename(os.path.normpath(folder_path))
326            new_sub_id = self._get_next_sub_id()
327            dest = os.path.join(self.workspace_path, new_sub_id)
328            shutil.copytree(folder_path, dest, dirs_exist_ok=True)
329
330            if not self._is_canceled:
331                self._rename_bids_files(dest, old_sub_id, new_sub_id)
332
333            log.debug(f"BIDS folder copied and renamed to {new_sub_id}.")  # <-- Log aggiornato
334            self.finished.emit()
335            return
336
337        # Case 2: Does the folder directly contain DICOM/NIfTI files (no subfolders)?
338        has_direct_medical_files = False
339        for file in os.listdir(folder_path):
340            file_path = os.path.join(folder_path, file)
341            if os.path.isfile(file_path):
342                if self._is_nifti_file(file) or self._is_dicom_file(file_path):
343                    has_direct_medical_files = True
344                    break
345
346        if self._is_canceled:
347            return
348
349        # If true, treat this as a single-patient folder
350        if has_direct_medical_files:
351            log.debug(f"Direct medical files found, processing as single patient: {folder_path}")
352            self._process_single_patient_folder(folder_path)
353            return
354
355        if self._is_canceled:
356            return
357
358        # Case 3: Folder has subdirectories â€” check what they represent
359        subfolders = [
360            os.path.join(folder_path, d)
361            for d in os.listdir(folder_path)
362            if os.path.isdir(os.path.join(folder_path, d))
363        ]
364        if not subfolders:
365            log.debug(f"No subfolders found in: {folder_path}")
366            return
367        if self._is_canceled:
368            return
369
370        # Check if all subfolders are DICOM series of the same patient
371        if self._are_dicom_series_of_same_patient(subfolders):
372            log.debug(f"DICOM series of same patient detected: {folder_path}")
373            self._process_single_patient_folder(folder_path)
374            return
375
376        if self._is_canceled:
377            return
378
379        # If subfolder names suggest multiple patients (e.g., 'sub-01', 'patient2'), process separately
380        if self._subfolders_look_like_different_patients(subfolders):
381            log.debug(f"Multiple patients detected by folder names: {folder_path}")
382            for subfolder in subfolders:
383                self._handle_import(subfolder)
384            return
385
386        if self._is_canceled:
387            return
388
389        # Fallback â€” treat as single-patient data
390        log.debug(f"Default: treating as single patient: {folder_path}")
391        self._process_single_patient_folder(folder_path)

394    def _are_dicom_series_of_same_patient(self, subfolders):
395        """
396        Check if multiple subfolders represent DICOM series from the same patient.
397
398        Heuristics:
399          - For each subfolder containing DICOM files, extract the PatientID or PatientName.
400          - If only one unique ID or name is found among all, we assume it's the same patient.
401
402        Args:
403            subfolders (list[str]): List of subfolder paths to inspect.
404
405        Returns:
406            bool: True if the subfolders seem to belong to the same patient.
407        """
408        log.debug(f"Checking if {len(subfolders)} subfolders are DICOM series of same patient...")
409
410        patient_ids = set()
411        dicom_folders_count = 0
412
413        for subfolder in subfolders:
414            log.debug(f"  Checking subfolder: {os.path.basename(subfolder)}")
415
416            has_dicom_in_subfolder = False
417            first_dicom_file = None
418
419            # Look for at least one DICOM file to sample metadata
420            for root, _, files in os.walk(subfolder):
421                for file in files:
422                    file_path = os.path.join(root, file)
423                    if self._is_dicom_file(file_path):
424                        has_dicom_in_subfolder = True
425                        first_dicom_file = file_path
426                        break
427                if has_dicom_in_subfolder:
428                    break
429
430            if self._is_canceled:
431                return
432
433            if has_dicom_in_subfolder:
434                dicom_folders_count += 1
435                log.debug(f"    Found DICOM files in: {os.path.basename(subfolder)}")
436
437                try:
438                    # Read DICOM metadata
439                    dcm = pydicom.dcmread(first_dicom_file, stop_before_pixels=True)
440                    patient_id = getattr(dcm, 'PatientID', None)
441                    patient_name = getattr(dcm, 'PatientName', None)
442                    masked_patient_name = f"{patient_name[:2]}...{patient_name[-2:]}" if patient_name else "unknown"
443
444                    log.debug(f"    Patient ID: {patient_id}, Patient Name: {masked_patient_name}")
445
446                    if patient_id:
447                        patient_ids.add(patient_id)
448                    elif patient_name:
449                        patient_ids.add(str(patient_name))
450
451                except Exception as e:
452                    log.debug(f"    Could not read DICOM metadata: {e}")
453            else:
454                log.debug(f"    No DICOM files found in: {os.path.basename(subfolder)}")
455
456        log.debug(f"Found {dicom_folders_count} folders with DICOM, {len(patient_ids)} unique patient IDs: {patient_ids}")
457
458        # Criteria: at least one DICOM folder, and at most one unique PatientID
459        is_same_patient = dicom_folders_count > 0 and len(patient_ids) <= 1
460        log.debug(f"  Result: is_same_patient = {is_same_patient}")
461
462        return is_same_patient

465    def _subfolders_look_like_different_patients(self, subfolders):
466        """
467        Heuristic that checks if subfolder names suggest multiple patients.
468
469        Example folder names that trigger 'multi-patient' assumption:
470            sub-01/, patient_1/, subj3/, p_01/, subject_2/
471
472        Args:
473            subfolders (list[str]): List of subfolder paths.
474
475        Returns:
476            bool: True if subfolder names suggest multiple subjects.
477        """
478        patient_like_prefixes = ['sub-', 'patient', 'subj', 'p_', 'subject']
479
480        for subfolder in subfolders:
481            folder_name = os.path.basename(subfolder).lower()
482            if any(folder_name.startswith(prefix) for prefix in patient_like_prefixes):
483                log.debug(f"Folder name '{folder_name}' suggests different patients")
484                return True
485
486        return False

574    def _convert_dicom_folder_to_nifti(self, dicom_folder, output_folder):
575        """
576        Converts a DICOM folder into NIfTI format using the external tool `dcm2niix`.
577
578        This function is executed for each single-patient folder that contains DICOM files.
579        It runs `dcm2niix` in a subprocess and places the resulting NIfTI and JSON metadata files
580        in the destination folder.
581
582        Args:
583            src_folder (str): The source folder containing DICOM files.
584            dest_folder (str): The folder where NIfTI files should be saved.
585        """
586        if os.path.isdir(output_folder):
587            for filename in os.listdir(output_folder):
588                if self._is_canceled:
589                    return
590                file_path = os.path.join(output_folder, filename)
591                try:
592                    if os.path.isfile(file_path) or os.path.islink(file_path):
593                        os.remove(file_path)
594                    elif os.path.isdir(file_path):
595                        shutil.rmtree(file_path)
596                except Exception as e:
597                    log.debug(f"Errore durante la rimozione di {file_path}: {e}")
598        else:
599            os.makedirs(output_folder, exist_ok=True)
600
601        if self._is_canceled:
602            shutil.rmtree(output_folder, ignore_errors=True)
603            return
604        self.current_progress += 10
605        self.progress.emit(self.current_progress)
606        try:
607            log.debug("DCM2NIIX path:" + get_bin_path("dcm2niix"))
608            command = [
609                get_bin_path("dcm2niix"),
610                "-f", "%p_%s",  # Naming format
611                "-p", "y",  # Preserve original acquisition order

300    def _handle_import(self, folder_path):
301        """
302        Handle the import of a single folder (which may itself contain subfolders).
303
304        This method performs heuristics to determine what kind of folder we are dealing with:
305          1. A BIDS folder â€” copy it directly.
306          2. A folder with direct medical files (NIfTI or DICOM) â€” process as one patient.
307          3. A folder containing subfolders â€” analyze if they belong to one or more patients.
308             - If subfolders are DICOM series of the same patient, process as one.
309             - If subfolders look like different patients (e.g., named 'sub-01', 'patient_02'), process each separately.
310             - Otherwise, default to single-patient mode.
311
312        Args:
313            folder_path (str): Path to the folder to process.
314        """
315        if not os.path.isdir(folder_path):
316            return
317        if self._is_canceled:
318            return
319
320        log.debug(f"Processing folder: {folder_path}")
321
322        # Case 1: The folder is already BIDS-organized -> just copy
323        if self._is_bids_folder(folder_path):
324            log.debug(f"BIDS structure detected in: {folder_path}")
325
326            old_sub_id = os.path.basename(os.path.normpath(folder_path))
327            new_sub_id = self._get_next_sub_id()
328            dest = os.path.join(self.workspace_path, new_sub_id)
329            shutil.copytree(folder_path, dest, dirs_exist_ok=True)
330
331            if not self._is_canceled:
332                self._rename_bids_files(dest, old_sub_id, new_sub_id)
333
334            log.debug(f"BIDS folder copied and renamed to {new_sub_id}.")  # <-- Log aggiornato
335            self.finished.emit()
336            return
337
338        # Case 2: Does the folder directly contain DICOM/NIfTI files (no subfolders)?
339        has_direct_medical_files = False
340        for file in os.listdir(folder_path):
341            file_path = os.path.join(folder_path, file)
342            if os.path.isfile(file_path):
343                if self._is_nifti_file(file) or self._is_dicom_file(file_path):
344                    has_direct_medical_files = True
345                    break
346
347        if self._is_canceled:
348            return
349
350        # If true, treat this as a single-patient folder
351        if has_direct_medical_files:
352            log.debug(f"Direct medical files found, processing as single patient: {folder_path}")
353            self._process_single_patient_folder(folder_path)
354            return
355
356        if self._is_canceled:
357            return
358
359        # Case 3: Folder has subdirectories â€” check what they represent
360        subfolders = [
361            os.path.join(folder_path, d)
362            for d in os.listdir(folder_path)
363            if os.path.isdir(os.path.join(folder_path, d))
364        ]
365        if not subfolders:
366            log.debug(f"No subfolders found in: {folder_path}")
367            return
368        if self._is_canceled:
369            return
370
371        # Check if all subfolders are DICOM series of the same patient
372        if self._are_dicom_series_of_same_patient(subfolders):
373            log.debug(f"DICOM series of same patient detected: {folder_path}")
374            self._process_single_patient_folder(folder_path)
375            return
376
377        if self._is_canceled:
378            return
379
380        # If subfolder names suggest multiple patients (e.g., 'sub-01', 'patient2'), process separately
381        if self._subfolders_look_like_different_patients(subfolders):
382            log.debug(f"Multiple patients detected by folder names: {folder_path}")
383            for subfolder in subfolders:
384                self._handle_import(subfolder)
385            return
386
387        if self._is_canceled:
388            return
389
390        # Fallback â€” treat as single-patient data
391        log.debug(f"Default: treating as single patient: {folder_path}")
392        self._process_single_patient_folder(folder_path)

395    def _are_dicom_series_of_same_patient(self, subfolders):
396        """
397        Check if multiple subfolders represent DICOM series from the same patient.
398
399        Heuristics:
400          - For each subfolder containing DICOM files, extract the PatientID or PatientName.
401          - If only one unique ID or name is found among all, we assume it's the same patient.
402
403        Args:
404            subfolders (list[str]): List of subfolder paths to inspect.
405
406        Returns:
407            bool: True if the subfolders seem to belong to the same patient.
408        """
409        log.debug(f"Checking if {len(subfolders)} subfolders are DICOM series of same patient...")
410
411        patient_ids = set()
412        dicom_folders_count = 0
413
414        for subfolder in subfolders:
415            log.debug(f"  Checking subfolder: {os.path.basename(subfolder)}")
416
417            has_dicom_in_subfolder = False
418            first_dicom_file = None
419
420            # Look for at least one DICOM file to sample metadata
421            for root, _, files in os.walk(subfolder):
422                for file in files:
423                    file_path = os.path.join(root, file)
424                    if self._is_dicom_file(file_path):
425                        has_dicom_in_subfolder = True
426                        first_dicom_file = file_path
427                        break
428                if has_dicom_in_subfolder:
429                    break
430
431            if self._is_canceled:
432                return
433
434            if has_dicom_in_subfolder:
435                dicom_folders_count += 1
436                log.debug(f"    Found DICOM files in: {os.path.basename(subfolder)}")
437
438                try:
439                    # Read DICOM metadata
440                    dcm = pydicom.dcmread(first_dicom_file, stop_before_pixels=True)
441                    patient_id = getattr(dcm, 'PatientID', None)
442                    patient_name = getattr(dcm, 'PatientName', None)
443                    masked_patient_name = f"{patient_name[:2]}...{patient_name[-2:]}" if patient_name else "unknown"
444
445                    log.debug(f"    Patient ID: {patient_id}, Patient Name: {masked_patient_name}")
446
447                    if patient_id:
448                        patient_ids.add(patient_id)
449                    elif patient_name:
450                        patient_ids.add(str(patient_name))
451
452                except Exception as e:
453                    log.debug(f"    Could not read DICOM metadata: {e}")
454            else:
455                log.debug(f"    No DICOM files found in: {os.path.basename(subfolder)}")
456
457        log.debug(f"Found {dicom_folders_count} folders with DICOM, {len(patient_ids)} unique patient IDs: {patient_ids}")
458
459        # Criteria: at least one DICOM folder, and at most one unique PatientID
460        is_same_patient = dicom_folders_count > 0 and len(patient_ids) <= 1
461        log.debug(f"  Result: is_same_patient = {is_same_patient}")
462
463        return is_same_patient

490    def _process_single_patient_folder(self, folder_path):
491        """
492        Process a folder assumed to contain data for a single patient.
493
494        This includes:
495          - Copying NIfTI files directly
496          - Converting DICOM series to NIfTI
497          - Organizing results into a BIDS-like folder structure
498          - Cleaning up temporary directories
499
500        Args:
501            folder_path (str): Path to the patient's folder.
502        """
503        if self._is_canceled:
504            return
505
506        log.debug(f"Processing single patient folder: {folder_path}")
507
508        nifti_files = []
509        dicom_files = []
510
511        base_folder_name = os.path.basename(os.path.normpath(folder_path))
512
513        # Temporary directory for conversion steps
514        temp_dir = tempfile.mkdtemp()
515        temp_base_dir = os.path.join(temp_dir, base_folder_name)
516
517        # Walk through the folder and collect relevant files
518        for root, _, files in os.walk(folder_path):
519            if self._is_canceled:
520                shutil.rmtree(temp_dir, ignore_errors=True)
521                return
522            for file in files:
523                if self._is_canceled:
524                    shutil.rmtree(temp_dir, ignore_errors=True)
525                    return
526
527                file_path = os.path.join(root, file)
528                relative_path = os.path.relpath(root, folder_path)
529                dest_dir = os.path.join(temp_base_dir, relative_path)
530                os.makedirs(dest_dir, exist_ok=True)
531
532                if self._is_nifti_file(file):
533                    nifti_files.append((file_path, os.path.join(dest_dir, file)))
534                elif self._is_dicom_file(file_path):
535                    dicom_files.append(file_path)
536                else:
537                    if self._is_canceled:
538                        return
539                    shutil.copy2(file_path, os.path.join(dest_dir, file))
540                    log.debug(f"Imported other file: {os.path.join(relative_path, file)}")
541
542        if self._is_canceled:
543            shutil.rmtree(temp_dir, ignore_errors=True)
544            return

546        # Copy NIfTI files to the temp structure
547        for src, dest in nifti_files:
548            if self._is_canceled:
549                shutil.rmtree(temp_dir, ignore_errors=True)
550                return
551            shutil.copy2(src, dest)
552            log.debug(f"Imported NIfTI file: {os.path.relpath(dest, temp_base_dir)}")
553
554        # If DICOMs exist, convert them into NIfTI using dcm2niix
555        if dicom_files:

559        if self._is_canceled:
560            shutil.rmtree(temp_dir, ignore_errors=True)
561            return
562
563        # Convert the temporary NIfTI/JSON folder into a BIDS structure
564        log.debug(f"Converting to BIDS structure...")
565        self._convert_to_bids_structure(temp_base_dir)
566
567        # Clean up
568        shutil.rmtree(temp_dir, ignore_errors=True)
569
570        # Optionally update UI buttons if context provides callback
571        if self.context and "update_main_buttons" in self.context:
572            self.context["update_main_buttons"]()
573        log.debug("Import completed for single patient.")

638    def _convert_to_bids_structure(self, input_folder):
639        """
640        Converts a folder containing NIfTI and JSON files into a BIDS-like directory structure.
641
642        The function scans all NIfTI files in the given folder and organizes them under:
643          derivatives/
644            â””â”€â”€ deep_learning_seg/
645                â””â”€â”€ sub-XXX/
646                    â””â”€â”€ anat/
647
648        Each new import increments the patient ID number (`sub-001`, `sub-002`, ...).
649
650        Args:
651            source_folder (str): Path to the folder containing NIfTI/JSON files to reorganize.
652        """
653        if self._is_canceled:
654            return
655        all_json_files = []
656        for root, _, files in os.walk(input_folder):
657            for file in files:
658                if file.endswith(".json"):
659                    all_json_files.append(os.path.join(root, file))
660
661        if not all_json_files:
662            log.debug("[BIDS] Nessun file JSON trovato. Conversione annullata.")
663            return
664
665        sub_id = self._get_next_sub_id()
666        dest_sub_dir = os.path.join(self.workspace_path, sub_id)
667        os.makedirs(dest_sub_dir, exist_ok=True)
668
669        pet_run_counter = 1
670        mr_run_counter = {}
671
672        for json_path in all_json_files:
673            nii_path = json_path.replace(".json", ".nii.gz")
674            if not os.path.exists(nii_path):
675                log.debug(f"[BIDS] Skipping: no matching NIfTI for {json_path}")
676                continue
677
678            with open(json_path, "r") as f:
679                metadata = json.load(f)
680
681            modality = metadata.get("Modality", "").upper()
682
683            if modality == "MR":
684                series_desc = metadata.get("ProtocolName", "").lower()
685
686                # Identifica il tipo (suffix BIDS)
687                if "flair" in series_desc:
688                    suffix = "flair"
689                elif "t1" in series_desc:
690                    suffix = "T1w"
691                elif "t2" in series_desc:
692                    suffix = "T2w"
693                else:
694                    suffix = "T1w"  # fallback
695
696                # Contatore run per stesso tipo
697                mr_run_counter.setdefault(suffix, 1)
698                run_label = f"run-{mr_run_counter[suffix]}"
699                mr_run_counter[suffix] += 1
700
701                # Directory anat
702                anat_dir = os.path.join(dest_sub_dir, "anat")
703                os.makedirs(anat_dir, exist_ok=True)
704
705                new_base = f"{sub_id}_{run_label}_{suffix}"
706
707                # Copia file
708                shutil.copy2(nii_path, os.path.join(anat_dir, f"{new_base}.nii.gz"))
709                shutil.copy2(json_path, os.path.join(anat_dir, f"{new_base}.json"))
710
711            elif modality == "PT":
712                # --- Ricava il tracciante ---
713                raw_trc = metadata.get("Radiopharmaceutical", "unknown")
714                # estrai la prima parola alfabetica come tracciante
715                trc = re.sub(r'[^a-zA-Z]', '', raw_trc.split()[0]).lower()
716                trc_label = f"trc-{trc}" if trc else "trc-unknown"
717
718                # --- Ricava il task ---
719                raw_task = metadata.get("SeriesDescription", "unknown")
720                task_clean = re.sub(r'[^a-zA-Z0-9]', '', raw_task.lower())
721                task_label = f"task-{task_clean}" if task_clean else "task-unknown"
722
723                # --- Determina se statico o dinamico ---
724                frame_durations = metadata.get("FrameDuration", [])
725                frame_times = metadata.get("FrameReferenceTime", [])
726                if (isinstance(frame_durations, (list, tuple)) and len(frame_durations) > 1) or \
727                        (isinstance(frame_times, (list, tuple)) and len(frame_times) > 1):
728                    acq_label = "rec-acdync"
729                    ses_label = "ses-02"
730                else:
731                    acq_label = "rec-acstat"
732                    ses_label = "ses-01"
733
734                pet_dir = os.path.join(dest_sub_dir, ses_label, "pet")
735                os.makedirs(pet_dir, exist_ok=True)
736
737                run_label = f"run-{pet_run_counter}"
738
739                # Add:
740                # trc_label for the tracer
741                # acq_label for the acquisition type (dynamic or static)
742                new_base = f"{sub_id}_{task_label}_{run_label}_pet"
743
744                shutil.copy2(nii_path, os.path.join(pet_dir, f"{new_base}.nii.gz"))
745                shutil.copy2(json_path, os.path.join(pet_dir, f"{new_base}.json"))
746
747                pet_run_counter += 1
748
749        log.debug(f"[BIDS] Imported {sub_id} in BIDS structure.")

751    def _is_bids_folder(self, folder_path):
752        """
753        Detects whether a folder already follows a BIDS structure.
754
755        Heuristic check:
756          - Folder contains 'sub-' directories.
757          - Inside those, there are 'anat', 'func', or 'dwi' subdirectories with NIfTI files.
758

760            folder_path (str): Path to check.
761
762        Returns:
763            bool: True if the folder appears to be BIDS-compatible.
764        """

766

769            return False
770
771        # Cerca sottocartelle BIDS standard con file utili
772        for session_or_modality in os.listdir(folder_path):

774            if os.path.isdir(session_path):
775                for modality in os.listdir(session_path):
776                    modality_path = os.path.join(session_path, modality)
777                    if os.path.isdir(modality_path):
778                        if any(f.endswith(".nii") or f.endswith(".nii.gz") for f in os.listdir(modality_path)):
779                            return True
780                # oppure Ã¨ una cartella "anat", "func", ecc. direttamente sotto sub-01 (senza ses-*)
781                if any(f.endswith(".nii") or f.endswith(".nii.gz") for f in os.listdir(session_path)):
782                    return True
783
784        return False

786    def _get_next_sub_id(self):
787        """
788        Generates the next available subject ID for the BIDS dataset.
