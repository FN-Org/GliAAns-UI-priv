85 def test_apply_overlay_numba(self): 

86 rgba_image = np.zeros((10, 10, 4), dtype=np.float64) 

87 overlay_mask = np.zeros((10, 10), dtype=bool) 

88 overlay_mask[5, 5] = True

91 

92 result = apply_overlay_numba(rgba_image, overlay_mask, overlay_intensity, overlay_color) 

93 

94 self.assertEqual(result[5, 5, 0], 0.5, "Red channel should reflect overlay intensity") 

95 self.assertEqual(result[5, 5, 1], 0.0, "Green channel should be zero for red overlay") 

96 self.assertEqual(result[5, 5, 2], 0.0, "Blue channel should be zero for red overlay") 

97 self.assertEqual(result[5, 5, 3], 0.0, "Alpha channel should remain zero") 

98 self.assertEqual(np.sum(result[:, :, 0] > 0), 1, "Only one pixel should be affected") 

99 

100 overlay_intensity = np.ones((10, 10)) 

101 result = apply_overlay_numba(rgba_image, overlay_mask, overlay_intensity, overlay_color) 

102 self.assertEqual(result[5, 5, 0], 1.0, "Red channel should be fully saturated") 

103 self.assertEqual(result[5, 5, 1], 0.0, "Green channel should be zero") 

104 self.assertEqual(result[5, 5, 2], 0.0, "Blue channel should be zero") 

105 

106 overlay_mask = np.zeros((10, 10), dtype=bool) 

107 result = apply_overlay_numba(rgba_image, overlay_mask, overlay_intensity, overlay_color) 

108 self.assertTrue(np.all(result == rgba_image), "No overlay should leave image unchanged") 

109 

110 rgba_image = np.zeros((10, 10, 4), dtype=np.float64) 

111 overlay_color = (0.0, 1.0, 0.0) 

112 overlay_intensity = np.ones((10, 10)) * 0.5 

113 overlay_mask[5, 5] = True 

114 result = apply_overlay_numba(rgba_image, overlay_mask, overlay_intensity, overlay_color)

116 self.assertEqual(result[5, 5, 0], 0.0, "Red channel should be zero for green overlay")

118 self.assertEqual(result[5, 5, 2], 0.0, "Blue channel should be zero for green overlay")
