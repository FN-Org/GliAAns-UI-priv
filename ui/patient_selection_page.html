334    def _select_all_patients(self):
335        """Mark all patients as selected."""
336        for patient_id, button in self.patient_buttons.items():
337            if not button.isChecked():
338                button.setChecked(True)
339                button.setText(QCoreApplication.translate("PatientSelectionFrame", "Selected"))
340                self.selected_patients.add(patient_id)
341        if self.context and "update_main_buttons" in self.context:
342            self.context["update_main_buttons"]()
343
344    def _deselect_all_patients(self):
345        """Unselect all patients."""
346        for patient_id, button in self.patient_buttons.items():
347            if button.isChecked():
348                button.setChecked(False)
349                button.setText(QCoreApplication.translate("PatientSelectionFrame", "Select"))
350                self.selected_patients.discard(patient_id)
351        if self.context and "update_main_buttons" in self.context:
352            self.context["update_main_buttons"]()
353
354    def _find_patient_dirs(self):
355        """Recursively find all patient directories (sub-*) in the workspace."""
356        patient_dirs = []
357        for root, dirs, _ in os.walk(self.workspace_path):
358            # Skip derivatives and pipeline folders
359            for skip in ("derivatives", "pipeline"):
360                if skip in dirs:
361                    dirs.remove(skip)
362
363            # Collect patient directories
364            for dir_name in dirs:
365                if dir_name.startswith("sub-"):
366                    patient_dirs.append(os.path.join(root, dir_name))
367
368            # Prevent recursion into patient subdirectories
369            dirs[:] = [d for d in dirs if not d.startswith("sub-")]
370
371        return patient_dirs
372
373    def _toggle_patient(self, patient_id, is_selected, button):
374        """Toggle selection for a single patient."""
375        if is_selected:
376            self.selected_patients.add(patient_id)
377            button.setText(QCoreApplication.translate("PatientSelectionFrame", "Selected"))
378        else:
379            self.selected_patients.discard(patient_id)
380            button.setText(QCoreApplication.translate("PatientSelectionFrame", "Select"))
381
382        if self.context and "update_main_buttons" in self.context:
383            self.context["update_main_buttons"]()
384
385    def get_selected_patients(self):
386        """Return a list of selected patient IDs."""
387        return list(self.selected_patients)

357        patient_dirs = []

360            for skip in ("derivatives", "pipeline"):

367                    patient_dirs.append(os.path.join(root, dir_name))
368
369            # Prevent recursion into patient subdirectories
370            dirs[:] = [d for d in dirs if not d.startswith("sub-")]
371
