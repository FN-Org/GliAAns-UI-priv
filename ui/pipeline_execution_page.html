 94    def _find_latest_config(self):
 95        """Find the config file with the highest numeric ID in the pipeline directory."""
 96        import glob
 97        pipeline_dir = os.path.join(self.workspace_path, "pipeline")
 98
 99        if not os.path.exists(pipeline_dir):
100            return os.path.join(pipeline_dir, "pipeline_config.json")
101
102        config_pattern = os.path.join(pipeline_dir, "*_config.json")
103        config_files = glob.glob(config_pattern)
104
105        if not config_files:
106            return os.path.join(pipeline_dir, "pipeline_config.json")
107
108        max_id = 0
109        latest_config = None
110
111        # Parse filenames to detect the latest config ID
112        for config_file in config_files:
113            filename = os.path.basename(config_file)
114            try:
115                config_id = int(filename.split('_')[0])
116                if config_id > max_id:
117                    max_id = config_id
118                    latest_config = config_file
119            except (ValueError, IndexError):
120                continue
121
122        return latest_config or os.path.join(pipeline_dir, "pipeline_config.json")

245    def _start_pipeline(self):
246        """Start the external pipeline process and monitor its output."""
247        if self.pipeline_process:
248            return  # Already running
249
250        self.pipeline_process = QProcess()
251        self.pipeline_process.finished.connect(self._on_process_finished)
252        self.pipeline_process.errorOccurred.connect(self._on_process_error)
253        self.pipeline_process.readyReadStandardOutput.connect(self._on_stdout_ready)
254        self.pipeline_process.readyReadStandardError.connect(self._on_stderr_ready)
255
256        self.progress_bar.setValue(0)
257        self.stop_button.setEnabled(True)
258        self._log_message(QCoreApplication.translate("PipelineExecutionPage", "Starting pipeline execution..."))
259
260        cmd = [
261            self.pipeline_bin_path,
262            "--config", self.config_path,
263            "--work-dir", self.workspace_path,
264            "--out-dir", self.pipeline_output_dir
265        ]
266        log.debug(f"{self.pipeline_bin_path} --config {self.config_path} --work-dir {self.workspace_path} --out-dir {self.pipeline_output_dir}")
267
268        self.pipeline_process.start(cmd[0], cmd[1:])
269
270        if not self.pipeline_process.waitForStarted(3000):
271            self._log_message(
272                QCoreApplication.translate("PipelineExecutionPage", "ERROR: Failed to start pipeline process"))
273            self._on_pipeline_error(
274                QCoreApplication.translate("PipelineExecutionPage", "Failed to start pipeline process"))

279    def _on_process_finished(self, exit_code, exit_status):
280        """Called when the process terminates."""
281        if exit_code == 0 and exit_status == QProcess.ExitStatus.NormalExit:
282            self._on_pipeline_finished()
283        else:
284            self._on_pipeline_error(
285                QCoreApplication.translate(
286                    "PipelineExecutionPage",
287                    "Process exited with code {exit_code}"
288                ).format(exit_code=exit_code)
289            )

291    def _on_process_error(self, error):
292        """Called when an error occurs in the process."""
293        error_messages = {
294            QProcess.ProcessError.FailedToStart: QCoreApplication.translate("PipelineExecutionPage",
295                                                                            "Failed to start process"),
296            QProcess.ProcessError.Crashed: QCoreApplication.translate("PipelineExecutionPage", "Process crashed"),
297            QProcess.ProcessError.Timedout: QCoreApplication.translate("PipelineExecutionPage", "Process timed out"),
298            QProcess.ProcessError.WriteError: QCoreApplication.translate("PipelineExecutionPage", "Write error"),

302        error_msg = error_messages.get(
303            error,
304            QCoreApplication.translate("PipelineExecutionPage",
305                                      "Process error: {error}").format(error=error)
306        )
307        self._on_pipeline_error(error_msg)

312    def _on_stdout_ready(self):
313        """Called when data is available on stdout."""
314        if self.pipeline_process:
315            data = self.pipeline_process.readAllStandardOutput()
316            output = data.data().decode('utf-8').strip()
317
318            for line in output.split('\n'):
319                if line.strip():
320                    self._process_pipeline_output(line.strip())
321
322    def _on_stderr_ready(self):
323        """Called when data is available on stderr."""
324        if self.pipeline_process:
325            data = self.pipeline_process.readAllStandardError()
326            error_output = data.data().decode('utf-8').strip()
327
328            for line in error_output.split('\n'):
329                if line.strip():
330                    self._log_message(f"STDERR: {line.strip()}")

333    # ─────────────────────────────────────────────
334    def _process_pipeline_output(self, line):
335        """Process the pipeline output logs"""
336        if line.startswith("LOG: "):
337            message = line[5:]
338            self._log_message(message)
339            self._update_current_operation(message)
340        elif line.startswith("ERROR: "):
341            error_msg = line[7:]
342            self._log_message(
343                QCoreApplication.translate("PipelineExecutionPage", "ERROR: {error}").format(error=error_msg))
344        elif line.startswith("PROGRESS: "):
345            progress_info = line[10:]
346            self._update_progress(progress_info)
347        elif line.startswith("PATIENT: "):
348            patient_info = line[9:]
349            self._log_message(
350                QCoreApplication.translate("PipelineExecutionPage", "Pipeline finished for patient: {message}").format(message=patient_info))
351            self._update_progress(patient_info)
352        elif line.startswith("FINISHED: "):
353            message = line[10:]
354            self._log_message(QCoreApplication.translate("PipelineExecutionPage", "FINISHED for: {message}").format(message=message))
355        else:
356            self._log_message(line)

395    def _update_progress(self, progress_info):
396        """
397        Update the circular progress bar based on progress information.
398
399        Expected format from pipeline: `"PROGRESS: X/Y"`
400
401        Args:
402            progress_info (str): Progress data string (e.g., `"3/10"`).
403        """
404        try:
405            if '/' in progress_info:
406                current, total = map(int, progress_info.split('/'))
407                percentage = int((current / total) * 100)
408                self.progress_bar.setValue(percentage)
409
410                self.check_new_files()  # Update folder views
411            elif "sub" in progress_info:
412                match = re.search(r"(sub-[a-zA-Z0-9_]+)", progress_info)
413                if match:
414                    sub_id = match.group(1)
415                    log.debug(f"Detected progress completion for: {sub_id}")
416
417                    for folder_path, card in self.folder_cards.items():
418                        folder_name = os.path.basename(folder_path)
419                        if sub_id in folder_name:
420                            log.debug(f"Setting finished state for card: {folder_name}")
421                            card.set_finished_state()
422                            break
423                    else:
424                        log.warning(f"No matching FolderCard found for {sub_id}")
425                else:
426                    log.warning(f"Could not extract sub-id from progress info: {progress_info}")
427
428        except ValueError:
429                log.warning("Failed to parse progress info")

431    def _update_current_operation(self, message):
432        """
433        Update the current operation label based on pipeline output message.
434
435        Args:
436            message (str): Message received from the pipeline log.

444            if patient_id:
445                self.current_operation.setText(
446                    QCoreApplication.translate(

448                        "Processing patient: {patient_id}"
449                    ).format(patient_id=patient_id)
450                )
451            else:
452                self.current_operation.setText(
453                    QCoreApplication.translate("PipelineExecutionPage", "Processing patient data...")
454                )
455        elif "analysis" in message.lower():
456            self.current_operation.setText(
457                QCoreApplication.translate("PipelineExecutionPage", "Performing statistical analysis...")
458            )
459        elif "saving" in message.lower() or "csv" in message.lower():
460            self.current_operation.setText(
461                QCoreApplication.translate("PipelineExecutionPage", "Saving results...")
462            )

464    def _extract_patient_id_from_log(self, message):
465        """
466        Extract a patient ID (e.g., 'sub-001') from a log message.
467
468        Args:
469            message (str): Log message.
470
471        Returns:
472            str | None: The patient ID if extracted, otherwise None.

474        import re

476        return f"sub-{match.group(1)}" if match else None

505    def _on_stop_clicked(self):
506        """
507        Handle a user click on the "Stop Pipeline" button.
508
509        Attempts graceful termination, followed by forced termination if necessary.
510        """
511        if self.pipeline_process and self.pipeline_process.state() == QProcess.ProcessState.Running:
512            self._log_message(QCoreApplication.translate("PipelineExecutionPage", "Stopping pipeline..."))
513
514            # Attempt graceful termination first
515            self.pipeline_process.terminate()
516
517            # Force termination if not stopped within 5 seconds
518            if not self.pipeline_process.waitForFinished(5000):
519                self.pipeline_process.kill()
520                self.pipeline_process.waitForFinished(3000)
521
522            self._log_message(QCoreApplication.translate("PipelineExecutionPage", "Pipeline stopped by user."))
523
524            self.current_operation.setText(
525                QCoreApplication.translate("PipelineExecutionPage", "Execution was interrupted by user."))
526
527            self.progress_bar.setValue(0)
528            self.stop_button.setEnabled(False)
529
530            self.pipeline_process = None
531            self.context["update_main_buttons"]()

 95    def _find_latest_config(self):
 96        """Find the config file with the highest numeric ID in the pipeline directory."""
 97        import glob
 98        pipeline_dir = os.path.join(self.workspace_path, "pipeline")
 99

104        config_files = glob.glob(config_pattern)

107            return os.path.join(pipeline_dir, "pipeline_config.json")

109        max_id = 0

112        # Parse filenames to detect the latest config ID
